Password Checker

import string def check_password_strength(password): strength = 0 remarks = '' lower_count = upper_count = num_count = wspace_count = special_count = 0 for char in password: if char in string.ascii_lowercase: lower_count += 1 elif char in string.ascii_uppercase: upper_count += 1 elif char in string.digits: num_count += 1 elif char == ' ': wspace_count += 1 else: special_count += 1 if lower_count >= 1: strength += 1 if upper_count >= 1: strength += 1 if num_count >= 1: strength += 1 if wspace_count >= 1: strength += 1 if special_count >= 1: strength += 1 if strength == 1: remarks = "Very Bad Password!!! Change ASAP" elif strength == 2: remarks = "Not A Good Password!!! Change ASAP" elif strength == 3: remarks = "It's a weak password, consider changing" elif strength == 4: remarks = "It's a hard password, but can be better" elif strength == 5: remarks = "A very strong password" return { 'lower_count': lower_count, 'upper_count': upper_count, 'num_count': num_count, 'wspace_count': wspace_count, 'special_count': special_count, 'strength': strength, 'remarks': remarks } def checkPassword(*args): password = Element('password').element.value result = check_password_strength(password) results_div = Element('results').element results_html = f"""
Lowercase characters: {result['lower_count']}
Uppercase characters: {result['upper_count']}
Numeric characters: {result['num_count']}
Whitespace characters: {result['wspace_count']}
Special characters: {result['special_count']}
Password Strength: {result['strength']}
Hint: {result['remarks']}
""" results_div.innerHTML = results_html def togglePasswordVisibility(): password_input = Element('password').element if password_input.type == 'password': password_input.type = 'text' else: password_input.type = 'password' # Bind the button click to the checkPassword function Element('check-btn').element.onclick = checkPassword
